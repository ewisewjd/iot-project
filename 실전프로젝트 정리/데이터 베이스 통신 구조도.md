    def check_id(self, input_id):
        conn, cursor = self.connect()  # DB 접속
        
        sql = """SELECT COUNT(*) FROM MEMBERS WHERE M_ID = :1"""
        cursor.execute(sql, (input_id,))  # 쿼리문 실행 
        data = cursor.fetchone()  # 조회 결과(한 튜플) 가져오기

        if data[0] == 0:
            return jsonify({'message':'success', 'result': True})
        else :
            return jsonify({'message':'success', 'result': False})
